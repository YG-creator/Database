2. CRUD
(0) 처음 설정
cd C:\bitnami\wamp\mysql\bin
서버접속 : mysql -uroot -p
DATABASE 생성 후 USE database이름;

(1) 생성
CREATE DATABASE 이름;
테이블 생성 : CREATE TABLE 이름(속성이름 datatype option PRIMARY KEY)
datatype : INT(),VARCHAR()
option : 빈칸 ㄴㄴ(NOT NULL), NULL, 자동증가(AUTO_INCREMNT)
PRIMARY KEY()
값 생성 : INSERT INTO TABLE이름 (속성1,속성2...) VALUES(값들 입력);
속성이름 바꾸기 : A AS B
(2) 읽기
SHOW DATABASES; 
SELECT * FROM TABLE이름;    :  전체 읽기
SELECT 원하는속성 FROM TABLE이름;   : 원하는 속성만 읽기
WHERE 속성이름 = 값  : 원하는 행만 읽기
ORDER BY 속성이름 DESC or ASC    : 정렬
LIMIT 숫자 :  몇개까지
(3) UPDATE
UPDATE TABLE이름 SET 바꿀 속성=원하는값, WHERE ~~;
(4)삭제
DROP FROM TABLE이름 WHERE 속성=값;
